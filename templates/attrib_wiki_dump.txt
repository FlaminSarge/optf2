$def with (pack)
{|class="wikitable grid sortable plainlinks"
|-
!class="header"|ID
!class="header"|Name
!class="header"|Description
!class="header"|Value Type
!class="header"|Class
$for attr in pack.get_item_schema_attributes():
    $ desc = pack.get_attribute_description(attr)
    $ name = pack.get_attribute_name(attr)
    $ attrtypefmt = "[[File:Pictogram {0}.png|16px]]"
    $ attrtypeimg = attrtypefmt.format("neutral")
    $ attrtype = pack.get_attribute_type(attr)
    $if attrtype == "negative": $ attrtypeimg = attrtypefmt.format("minus")
    $elif attrtype == "positive": $ attrtypeimg = attrtypefmt.format("plus")
    $if desc: $ desc = desc.replace("\n", "<br/>").replace("%s1", "n")
    |-
    |$pack.get_attribute_id(attr)
    |$attrtypeimg [${instance.homedomain}$qurl(att = name) $name]
    |<nowiki>$desc</nowiki>
    |$pack.get_attribute_value_type(attr)
    |$pack.get_attribute_class(attr)
|}
